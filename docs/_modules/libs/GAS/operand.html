[docs]    def randomize(self) -> None:
        for op in self.all.values():
            op.val = op.val if op.is_imm else randint(-31, 31)

[docs]    def same_as(self, op_dict) -> bool:
        for key in self.all:
            if key not in op_dict.all:
                continue
            if self.all[key].val != op_dict.all[key].val:
                return False
        return True

    

[docs]    def add(self, op: Operand) -> None:
        if op.name not in self.all:
            self.all[op.name] = op

        

[docs]    def filter_self(self):
        """不要な変数を取り除く

        難読化によって自動生成された変数を除いたOperandDictを返す．

        """
        res = OperandDict()
        for op in self.all:
            if op.is_imm or op.is_dummy:
                continue
            res.add(op)
        return res
