[docs]def simulate(lines: list[Line], op_dict: OperandDict) -> list[OperandDict]:
    init = op_dict
    res  = init.copy()
    sim = Simulator(res)

    for i in range(100):
        try:
            line = lines[sim.pc]
        except:
            return init, res, sim
        sim.transform(line.tree)
        if type(sim.pc) is not int:
            sim.pc = find_label(lines, sim.pc)

    return init, res, sim
